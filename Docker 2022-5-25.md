## 安装

1. 卸载旧的版本

   ```
   sudo yum remove docker \
                     docker-client \
                     docker-client-latest \
                     docker-common \
                     docker-latest \
                     docker-latest-logrotate \
                     docker-logrotate \
                     docker-engine
   ```

   

2. 设置```yum-utils```包，（提供`yum-config-manager` 实用程序）并设置**稳定**的存储库。

```shell
 sudo yum install -y yum-utils

 # 设置远程仓库的地址可以用 http://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo 代替
 sudo yum-config-manager \
    --add-repo \
    https://download.docker.com/linux/centos/docker-ce.repo
```



3. 安装docker引擎

```
sudo yum install docker-ce docker-ce-cli containerd.io docker-compose-plugin
```



4. 启动docker

   ```
   sudo systemctl start docker
   ```



5. hello-world 测试 docker是否安装成功

   ```
   sudo docker run hello-world
   ```



## Docker 常用命令

> 命令帮助文档的地址： https://docs.docker.com/engine/reference

```shell
	命令								备注
	docker version					# docker 版本信息
	docker info  					# docker 详细信息，保活镜像和容器的数量
	docker 命令 --help		       # docker 帮助命令
```



### Docker 镜像命令



```shell
	命令											备注
	docker search 镜像名称						 				# 搜索镜像
	docker pull 镜像名[:TAG]					  				 # 下载镜像，不带tag默认下最新的
	docker images [-q只看ID]					   				  # 查看宿主机所有的镜像
    docker rmi 镜像ID				 			   				  # 删除指定镜像
    docker rmi -f 镜像ID			 			   				  # 强制删除（如果镜像正在运行）
    docker rmi -f 镜像ID:[TAG] 镜像ID:[TAG]		  			     # 删除多个镜像
    docker rmi $(docker images -qa) 			 			    # 删除所有镜像
    docker commit -m"描述信息" -a"作者" 容器id 镜像名称:[TAG]		# 提交镜像
```





### Docker 容器命令



```shell
	命令																	备注
	docker run -d --name 容器名称|容器ID -p 33306:3306 mysql:5.7	 		 # 创建容器
	-i 运行容器
	-t 容器启动后进入该容器执行命令
	-name 为创建的容器命名
	-d 后台运行
	-p 宿主机端口:容器端口，进行映射
	-P 随机端口
	-e username='cat'，设置环境变量
	-env-file=[] 从指定文件读入环境变量
	
	
	docker start 容器ID|name|name:tag										# 启动容器
	docker exec -it 容器ID|容器名称 /bin/bash								 # 进入容器
	exit																 # 退出容器
	docker ps  														     # 查看当前运行的容器
	docker ps -a 														 # 查看所有容器（包括没运行的）
	docker stop 容器ID 													# 平滑关闭
	docker kill 容器ID 													# 强制关闭
	docker restart 容器名称|容器ID										  # 重启容器
	docker pause 容器ID													# 暂停容器
	docker top 容器名|容器ID												  # 查看容器内运行的进程
	
	docker logs -t -f --tail 10  容器ID|容器名称							  # 查看容器日志
	-t 显示时间
	-f 追加查看
	–tail 看最后几条
	
	docker rm 容器名称|容器ID    										      # 删除容器，容器必须要关闭
	docker rm -f 容器名称|容器ID 强制删除容器（运行和未运行的）					# 删除容器，运行和未运行的都关闭
	docker rm -f $(docker ps -q)										  # 删除全部正在运行的容器
	docker start $(docker ps -aq -f status=exited)						  # 启动所有状态为 exited的容器
	docker start $(docker ps -aq) 										  # 启动所有容器
	docker cp 容器ID|容器名称:容器内路径 目的地目录								# 容器内文件拷贝到宿主机
```

## 容器数据卷

1. 具名挂载

> 将Docker容器中的数据文件同步到宿主机上

```shell
 # 需要在dokcer run 命令的时候指定
 docker run -it -v /宿主机绝对路径目录:/容器内目录[:ro|:rw]      镜像名|镜像ID
 ro readonly # 只读
 rw readwrite # 可读可写
 
 # 查看 挂载情况
 docker volume inspect 容器名|容器ID
```



2. 匿名挂载

   > 执行 docker run 命令时，没有指定 容器外路径 ，那么就是匿名挂载，挂载的数据统一放在`/var/lib/docker`下

## 卸载

1. 卸载 Docker Engine、CLI、Containerd 和 Docker Compose 软件包

```
sudo yum remove docker-ce docker-ce-cli containerd.io
```



2. 主机上的映像、容器、卷或自定义配置文件不会自动删除。要删除所有映像、容器和卷

```
sudo rm -rf /var/lib/docker
sudo rm -rf /var/lib/containerd
```
